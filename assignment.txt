19.Write pseudocode for implementing concept of menu driven calculator which will accept two numbers and perform the calculation based upon users choice like(+,-,*,/).

using System;

public class Calculator
{
    public static void Main(string[] args)
    {
      int choice;
      //float num1,num2,ans;
       Console.WriteLine("Enter the first number");
      float num1=float.Parse(Console.ReadLine());
      Console.WriteLine("Enter the second number");
      float num2=float.Parse(Console.ReadLine());
      Console.WriteLine("Enter the operation");
      Console.WriteLine("+ addition");
      Console.WriteLine("- subtraction");
      Console.WriteLine("* multiplication");
      Console.WriteLine("/ division");
      choice=Convert.ToChar(Console.ReadLine());
      
        
        float ans;
      switch(choice)
      {
       
        
          case '+' :ans=num1+num2;
                 Console.WriteLine($"ans={ans}");
                 break;
          case '-' :ans=num1-num2;
                 Console.WriteLine($"ans={ans}");
                 break;
          case '*' :ans=num1*num2;
                 Console.WriteLine($"ans={ans}");
                 break;
          case '/' :ans=num1/num2;
                 Console.WriteLine($"ans={ans}");
                 break;

          default: Console.WriteLine("Invalid operation");
                 break;
      }
      
       
    }
}
